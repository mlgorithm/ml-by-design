\begin{tikzpicture}[

%% ── Pre-registration ─────────────────────────────────────────────────────────
\node[fstep, fill=claimcolor!10, draw=claimcolor] (pre) at (5.0, 9.2)

%% ── Step 1 ───────────────────────────────────────────────────────────────────
\node[fstep, fill=claimcolor!10, draw=claimcolor] (s1) at (5.0, 7.7)
    {\textcolor{claimcolor}{\bfseries 1.\enspace State the Claim}\enspace —
     one sentence, specific scope; name the weaker and stronger alternatives};

%% ── Step 2 ───────────────────────────────────────────────────────────────────
\node[fstep, fill=protectcolor!10, draw=protectcolor] (s2) at (5.0, 6.2)
    {\textcolor{protectcolor}{\bfseries 2.\enspace Protect Protocol}\enspace —
     freeze the test set; guard against leakage, distribution shift, and temporal ordering};

%% ── Step 3 ───────────────────────────────────────────────────────────────────
\node[fstep, fill=controlcolor!10, draw=controlcolor] (s3) at (5.0, 4.7)
    {\textcolor{controlcolor}{\bfseries 3.\enspace Choose Control}\enspace —
     use a strong, domain-appropriate baseline tuned with equal care};

%% ── Step 4 ───────────────────────────────────────────────────────────────────
\node[fstep, fill=uncertaincolor!10, draw=uncertaincolor] (s4) at (5.0, 3.2)
    {\textcolor{uncertaincolor}{\bfseries 4.\enspace Quantify Uncertainty}\enspace —

%% ── Step 5 ───────────────────────────────────────────────────────────────────
\node[fstep, fill=mechanismcolor!10, draw=mechanismcolor] (s5) at (5.0, 1.7)
    {\textcolor{mechanismcolor}{\bfseries 5.\enspace Isolate Mechanism}\enspace —

%% ── Step 6 ───────────────────────────────────────────────────────────────────
\node[fstep, fill=failurecolor!10, draw=failurecolor] (s6) at (5.0, 0.2)
    {\textcolor{failurecolor}{\bfseries 6.\enspace Inspect Failures}\enspace —

%% ── Step 7 ───────────────────────────────────────────────────────────────────
\node[fstep, fill=scopecolor!10, draw=scopecolor] (s7) at (5.0, -1.3)
    {\textcolor{scopecolor}{\bfseries 7.\enspace Bound the Scope}\enspace —

    {\textcolor{conclusioncolor}{Bounded Scientific Conclusion}};
