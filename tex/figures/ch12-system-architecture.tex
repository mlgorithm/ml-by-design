\begin{tikzpicture}[
  font=\footnotesize,
  service/.style={draw, rounded corners, align=center, minimum width=2.7cm, minimum height=0.8cm, fill=blue!10},
  store/.style={draw, cylinder, shape border rotate=90, align=center, minimum width=1.65cm, minimum height=0.7cm, fill=yellow!18},
  ops/.style={draw, rounded corners, align=center, minimum width=2.35cm, minimum height=0.8cm, fill=red!12},
  ext/.style={draw, align=center, minimum width=2.1cm, minimum height=0.7cm, fill=gray!15},
  arrow/.style={-Latex, thick, black!65},
  dataflow/.style={-Latex, thick, blue!65!black},
  feedback/.style={-Latex, thick, dashed, red!70!black}
]

% Core service path
\node[ext]     (user)     at (0,6.6) {User request};
\node[service] (contract) at (0,5.2) {Input contract\\and validation};
\node[service] (context)  at (0,3.8) {Build served context\\or retrieve evidence};
\node[service] (model)    at (0,2.4) {Model score\\or answer draft};
\node[service] (policy)   at (0,1.0) {Policy, threshold,\\and escalation rule};
\node[ext]     (auto)     at (-2.8,-0.6) {Automated\\response};
\node[ext]     (review)   at (2.8,-0.6) {Human review\\queue};

% Data stores
\node[store]   (sources)  at (-4.4,3.8) {Approved\\sources};
\node[store]   (rules)    at (4.4,1.0) {Policy\\rules};
\node[store]   (logs)     at (4.4,3.0) {Event\\log};

% Operations lane
\node[ops]     (monitor)  at (7.0,3.0) {Monitoring\\and alerts};
\node[ext]     (owner)    at (7.0,1.0) {Owner and\\rollback};

% Improvement loop
\node[service] (feedbackbox) at (-3.0,1.0) {Feedback, overrides,\\and outcomes};
\node[service] (update)      at (-4.4,-0.6) {Model or policy\\update};

% Main flow
\draw[arrow] (user) -- (contract);
\draw[arrow] (contract) -- (context);
\draw[arrow] (context) -- (model);
\draw[arrow] (model) -- (policy);
\draw[arrow] (policy) -- (auto);
\draw[arrow] (policy) -- (review);

% Evidence and rule access
\draw[dataflow] (sources) -- (context);
\draw[dataflow] (rules) -- (policy);

% Logging and operations
\draw[dataflow, rounded corners=4pt] (contract.east) -| (logs);
\draw[dataflow, rounded corners=4pt] (model.east) -| (logs);
\draw[dataflow, rounded corners=4pt] (policy.east) -| (logs);
\draw[dataflow] (logs) -- (monitor);
\draw[arrow] (monitor) -- (owner);

% Feedback loops
\draw[feedback] (review) -- (feedbackbox);
\draw[feedback] (auto.west) to[out=180,in=20] (feedbackbox.north);
\draw[feedback] (feedbackbox) -- (update);
\draw[feedback] (owner.west) -- (policy.east);
\draw[feedback] (owner.west) to[out=165,in=0] (contract.east);

% Lane labels
\node[font=\tiny, gray!70] at (-4.9,6.0) {\textbf{inputs}};
\node[font=\tiny, blue!65!black] at (0.8,6.0) {\textbf{service path}};
\node[font=\tiny, red!70!black] at (7.1,4.1) {\textbf{operations}};
\node[font=\tiny, black!65] at (-3.9,1.9) {\textbf{improvement loop}};

\end{tikzpicture}
