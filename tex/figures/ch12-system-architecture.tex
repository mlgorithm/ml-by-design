\begin{tikzpicture}[
  font=\footnotesize,
  component/.style={draw, rectangle, rounded corners, align=center, minimum width=2.0cm, minimum height=0.8cm, fill=blue!10},
  datastore/.style={draw, cylinder, shape border rotate=90, align=center, minimum width=1.5cm, minimum height=0.6cm, fill=yellow!15},
  external/.style={draw, rectangle, align=center, minimum width=1.8cm, minimum height=0.6cm, fill=gray!15},
  monitor/.style={draw, rectangle, rounded corners, align=center, minimum width=1.8cm, minimum height=0.6cm, fill=red!15},
  arrow/.style={-Latex, thick},
  dataflow/.style={-Latex, thick, blue},
  feedback/.style={-Latex, dashed, red},
  alertflow/.style={-Latex, thick, red}
]

% Input Layer
\node[external] (user) at (0,8) {Student\\Message};
\node[component] (validate) at (0,6.5) {Input Contract\\\& Validation};

% Core Pipeline
\node[component] (preprocess) at (0,5) {Preprocessing\\\& Feature Eng};
\node[component] (model) at (0,3.5) {Model\\Inference};
\node[component] (threshold) at (0,2) {Threshold\\\& Policy};

% Decision Layer
\node[component] (decide) at (0,0.5) {Route Decision};
\node[external] (autoresponse) at (-2.5,0.5) {Automated\\Response};
\node[external] (humanreview) at (3,0.5) {Human\\Review Queue};

% Data Stores
\node[datastore] (knowledge) at (-4.5,3.5) {Knowledge\\Base};
\node[datastore] (policies) at (4,2.5) {Policies \&\\Rules};
\node[datastore] (logs) at (2,5) {Event\\Logs};

% Monitoring & Control
\node[monitor] (monitor) at (6,6) {Monitoring\\\& Alerts};
\node[monitor] (dashboard) at (6,4) {Performance\\Dashboard};
\node[external] (owner) at (6,1) {System\\Owner};

% Feedback & Control Loops
\node[component] (feedback) at (-2.5,2) {Feedback\\Collection};
\node[component] (retraining) at (-4.5,0.5) {Model\\Updates};

% Main forward flow
\draw[arrow] (user) -- (validate);
\draw[arrow] (validate) -- (preprocess);
\draw[arrow] (preprocess) -- (model);
\draw[arrow] (model) -- (threshold);
\draw[arrow] (threshold) -- (decide);
\draw[arrow] (decide) -- (autoresponse);
\draw[arrow] (decide) -- node[above, font=\tiny] {Escalate} (humanreview);

% Data access
\draw[dataflow] (knowledge) -- (model);
\draw[dataflow] (policies) -- (threshold);

% Logging flows - right-angle routing keeps arrows east of pipeline boxes
\draw[dataflow, rounded corners=5pt] (validate.east) -| (logs);
\draw[dataflow, rounded corners=5pt] (model.east) -| (logs);

% Monitoring flows
\draw[dataflow] (logs) -- (monitor);
\draw[alertflow] (monitor) -- (dashboard);
\draw[alertflow] (dashboard) -- (owner);

% Control feedback loops
\draw[feedback] (owner.west) -- (threshold.east);
\draw[feedback] (feedback) -- (retraining);
\draw[feedback] (autoresponse) to[out=120,in=240] (feedback);

% Rollback capability
\draw[feedback, very thick] (owner) to[out=150,in=60] (decide);

% Labels for system boundaries
\node[font=\tiny, gray] at (-5.5,7) {\textbf{External}};
\node[font=\tiny, blue] at (1,7.4) {\textbf{Core Pipeline}};
\node[font=\tiny, red] at (7,7) {\textbf{Operations}};

\end{tikzpicture}
